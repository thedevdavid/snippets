## Install

```bash
npx shadcn@latest add https://snippets.thedevdavid.com/r/next-supabase-subdomain-cookies.json
```

## Usage

```tsx
import { createServerClient } from "@/lib/cookies";

export async function GET(request: NextRequest) {
  const cookieStore = await createServerClient(request);
  const { data } = await cookieStore.get("user");
}
```
